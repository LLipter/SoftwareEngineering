\section*{Software Metrics}

\begin{itemize}

	\item \textbf{Size}

	In lines of code, or better
	
	\item \textbf{Cost}
	
	In dollars
	
	\item \textbf{Duration}
	
	In months
	
	\item \textbf{Effort}
	
	In person months
	
	\item \textbf{Quality}
	
	Number of faults detected

\end{itemize}
